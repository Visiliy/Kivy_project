def build(self):
    fl = FloatLayout(size=(300, 300))
    fl.add_widget(Button(text='Кнопка', font_size='40', on_press=self.btn_press, background_color=[1, 0, 0, 1],
                         background_normal='', size_hint=(.25, .15), pos=(400, 300)))
    return fl
def btn_press(self, instance):
    print('Кнопка нажата')
    instance.text = 'Hello'
